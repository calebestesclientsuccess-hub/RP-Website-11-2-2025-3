/* 1. GRADIENT TEXT SYSTEM
  Apply this class to any h1, h2, or key text element.
*/
.gradient-text {
  background: linear-gradient(
    90deg,
    #ff6b6b 0%,  /* Coral start */
    #ff6b9d 20%, /* Pink transition */
    #c44b8c 40%, /* Magenta */
    #9f8fff 60%, /* Your brand purple */

    #4a90e2 100% /* Blue end */
  );
  -webkit-background-clip: text;
  background-clip: text;
  -webkit-text-fill-color: transparent;
  /* Fallback for browsers that don't support background-clip */
  color: #9f8fff;
}

/* 2. TECHNICAL GRID PATTERN LAYERS
  Apply these to backgrounds or card pseudo-elements.
*/
/* Dot grid - engineering paper style */
.light-grid-dots {
  background-image: radial-gradient(
    circle at center,
    hsla(var(--foreground), 0.03) 1px,
    transparent 1px
  );
  background-size: 24px 24px;
  background-position: 0 0, 12px 12px;
}

/* Isometric grid for hero sections */
.light-grid-iso {
  background-image: linear-gradient(
      30deg,
      hsla(var(--border), 0.3) 1px,
      transparent 1px
    ),
    linear-gradient(-30deg, hsla(var(--border), 0.3) 1px, transparent 1px);
  background-size: 40px 69.28px; /* Perfect isometric ratio */
}

/* 3. DEPTH THROUGH LIGHT PHYSICS
  Apply this class to your light-mode Shadcn <Card>
*/
.light-depth-card {
  box-shadow:
    /* Ambient occlusion - subtle self-shadow */
      inset 0 -1px 2px -1px hsla(var(--foreground), 0.04),
    /* Contact shadow - grounding */ 0 1px 3px -1px hsla(var(--foreground), 0.06),
    /* Soft elevation */ 0 2px 8px -4px hsla(var(--foreground), 0.08);
  transition: box-shadow 0.3s ease;
}

/* Optional: Fresnel effect on hover - edges catch light */
.light-depth-card:hover::before {
  content: '';
  position: absolute;
  inset: 0;
  border-radius: inherit; /* Assumes it's on a rounded element */
  background: linear-gradient(
    135deg,
    transparent 40%,
    hsla(0, 0%, 100%, 0.4) 50%,
    transparent 60%
  );
  opacity: 1;
  animation: light-sweep 0.6s ease-out;
}

/* 5. MICRO-ANIMATIONS SUITE
  Define the keyframes. We'll apply these with utility classes or GSAP.
*/

/* Animation for the Fresnel hover effect */
@keyframes light-sweep {
  from {
    opacity: 0;
    transform: translateX(-50%);
  }
  to {
    opacity: 1;
    transform: translateX(0%);
  }
}

/* Text gradient reveal on scroll */
@keyframes gradient-reveal {
  from {
    background-size: 0% 100%;
    -webkit-text-fill-color: hsl(var(--foreground));
  }
  to {
    background-size: 100% 100%;
    -webkit-text-fill-color: transparent;
  }
}

/* Number counter with shimmer */
/* Apply to a .gradient-text element with .animate-shimmer */

  0% {
    background-position: 200% center;
  }
  100% {

  }

/* Light scanning effect for progress bars */

  }
  100% {
    box-shadow: inset 200px 0 0 0 hsla(var(--primary), 0.2);
  }
}

/* 6. REDUCED MOTION PREFERENCES
  This ensures accessibility by disabling animations for users who prefer it.
*/
@media (prefers-reduced-motion: reduce) {
  .light-depth-card:hover::before,
  .animate-shimmer,
  [data-animation='gradient-reveal'] {
    animation: none;
    transition: none;
  }
}
