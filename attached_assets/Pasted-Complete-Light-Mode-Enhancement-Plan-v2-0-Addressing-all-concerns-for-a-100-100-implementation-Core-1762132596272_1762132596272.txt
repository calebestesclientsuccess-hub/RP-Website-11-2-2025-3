2. Performance-Aware Effect Tiers
// Performance detection system
function detectPerformanceTier() {
  const checks = {
    gpu: checkWebGLPerformance(),
    memory: navigator.deviceMemory || 4,
    cores: navigator.hardwareConcurrency || 2,
    connection: navigator.connection?.effectiveType || '4g',
    battery: navigator.getBattery ? await navigator.getBattery() : null
  };
  
  // Tier 1: Basic (gradient text only)
  // Tier 2: Enhanced (+ grid patterns, basic animations)  
  // Tier 3: Full (all effects including light physics)
  
  if (checks.memory < 4 || checks.cores < 4 || checks.connection === '2g') {
    return 1;
  } else if (checks.memory < 8 || !checks.gpu.highPerformance) {
    return 2;
  }
  return 3;
}

// Smooth theme toggle
function toggleTheme() {
  document.documentElement.classList.add('theme-transitioning');
  
  // Fade out gradients
  document.querySelectorAll('.gradient-mesh-layer').forEach(el => {
    el.style.opacity = '0';
  });
  
  setTimeout(() => {
    // Switch theme
    document.documentElement.classList.toggle('dark');
    
    // Fade in new gradients
    setTimeout(() => {
      document.querySelectorAll('.gradient-mesh-layer').forEach(el => {
        el.style.opacity = '';
      });
      document.documentElement.classList.remove('theme-transitioning');
    }, 50);
  }, 200);
}
